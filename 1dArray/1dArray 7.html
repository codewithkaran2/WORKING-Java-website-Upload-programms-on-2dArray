class main {
        public static void main(String st[]) {
            int arr[];
            arr = new int[8];
    
            arr[0] = 3;
            arr[1] = 96;
            arr[2] = 112;
            arr[3] = 8;
            arr[4] = 91;
            arr[5] = 6;
            arr[6] = 41;
            arr[7] = 0;
    
            System.out.println("Original array is:");
    
            for (int i = 0; i < 8; i++) {
                System.out.print(arr[i] + " ");
            }
    
            System.out.println("\nArray elements at even position is:");
    
            for (int i = 0; i < 8; i++) {
                if (i % 2 == 0)
                    System.out.print(arr[i] + " ");
            }
        }
    }
