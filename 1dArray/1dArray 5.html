//Program to print the elements of an array

    class main {
        public static void main(String st[]) {
            int arr[];
            arr = new int[5];
    
            arr[0] = 1;
            arr[1] = 2;
            arr[2] = 3;
            arr[3] = 4;
            arr[4] = 5;
    
            System.out.println("array is:");
    
            for (int i = 0; i < 5; i++) {
                System.out.print(arr[i] + " ");
            }
        }
    }
