//Program to copy all elements of one array into another array

class main {

    public static void main(String st[]) {

        int arr[], arr2[];
        arr = new int[5];
        arr2 = new int[5];

        arr[0] = 23;
        arr[1] = 56;
        arr[2] = 12;
        arr[3] = 87;
        arr[4] = 9;

        for (int i = 0; i < 5; i++) {
            arr2[i] = arr[i];
        }

        System.out.println("Copied array is:");

        for (int i = 0; i < 5; i++) {
            System.out.print(arr2[i] + " ");
        }
    }
}
