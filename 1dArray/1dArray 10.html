class main {

        public static void main(String st[]) {
    
            int arr[], min;
            arr = new int[8];
    
            arr[0] = 3;
            arr[1] = 96;
            arr[2] = 112;
            arr[3] = 8;
            arr[4] = 91;
            arr[5] = 6;
            arr[6] = 41;
            arr[7] = 0;
    
            System.out.println("Original array is:");
    
            for (int i = 0; i < 8; i++) {
                System.out.print(arr[i] + " ");
            }
    
            min = arr[0];
    
            System.out.println("\nSmallest element in array is:");
    
            for (int i = 1; i < 8; i++) {
                if (min > arr[i])
                    min = arr[i];
            }
    
            System.out.print(min);
        }
    }
