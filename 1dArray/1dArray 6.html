//Program to print the elements of an array in reverse order

    class ReverseArray {
        public static void main(String[] args) {
            int[] array = {1, 2, 3, 4, 5};
            
            for (int i = array.length - 1; i >= 0; i--) {
                System.out.println(array[i]);
            }
        }
    }
